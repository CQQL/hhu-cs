    \sum_{k = 1}^{j - 1} 2^{k - 1} (j - k) & = 2^{j - 2} (j - j + 1) + \sum_{k = 1}^{j - 2} 2^{k - 1} (j - 1 - k) + \sum_{k = 1}^{j - 2} 2^{k - 1}\\
    & = 2^{j - 2} + 2^{j - 1} - j + \sum_{k = 1}^{j - 2} 2^{k - 1}\\
    & = 2^{j - 2} + 2^{j - 1} - j + 2^{j - 2} - 1\\

  \end{align*}
